# az k8s-runtime

```bash
# Create a BfdProfile.
az k8s-runtime bfd-profile create

# Delete a BfdProfile.
az k8s-runtime bfd-profile delete

# List BfdProfile resources by parent.
az k8s-runtime bfd-profile list

# Get a BfdProfile.
az k8s-runtime bfd-profile show

# Update a BfdProfile.
az k8s-runtime bfd-profile update

# Place the CLI in a waiting state until a condition is met.
az k8s-runtime bfd-profile wait

# Create a BgpPeer.
az k8s-runtime bgp-peer create

# Delete a BgpPeer.
az k8s-runtime bgp-peer delete

# List BgpPeer resources by parent.
az k8s-runtime bgp-peer list

# Get a BgpPeer.
az k8s-runtime bgp-peer show

# Update a BgpPeer.
az k8s-runtime bgp-peer update

# Place the CLI in a waiting state until a condition is met.
az k8s-runtime bgp-peer wait

# Create a LoadBalancer.
az k8s-runtime load-balancer create

# Delete a LoadBalancer.
az k8s-runtime load-balancer delete

# Disable load_balancer service in a connected cluster.
az k8s-runtime load-balancer disable

# Enable load balancer service in a connected cluster.
az k8s-runtime load-balancer enable

# List LoadBalancer resources by parent.
az k8s-runtime load-balancer list

# Get a LoadBalancer.
az k8s-runtime load-balancer show

# Update a LoadBalancer.
az k8s-runtime load-balancer update

# Place the CLI in a waiting state until a condition is met.
az k8s-runtime load-balancer wait
```
