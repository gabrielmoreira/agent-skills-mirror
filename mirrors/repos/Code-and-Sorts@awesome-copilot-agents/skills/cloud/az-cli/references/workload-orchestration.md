# az workload-orchestration

```bash
# Create a Config Template Resource.
az workload-orchestration config-template create

# Delete a Config Template Resource.
az workload-orchestration config-template delete

# Link a Config Template to hierarchies.
az workload-orchestration config-template link

# List by subscription.
az workload-orchestration config-template list

# Remove Config Template Version Resource.
az workload-orchestration config-template remove-version

# Get a Config Template Resource.
az workload-orchestration config-template show

# Unlink a Config Template from hierarchies.
az workload-orchestration config-template unlink

# Place the CLI in a waiting state until a condition is met.
az workload-orchestration config-template wait

# Show linked hierarchies for a config template.
az workload-orchestration config-template hierarchy show

# List Config Template Version Resources.
az workload-orchestration config-template version list

# Get a Config Template Version Resource.
az workload-orchestration config-template version show

# Download configurations available at specified hierarchical entity.
az workload-orchestration configuration download

# To set the values to configurations available at specified hierarchical entity.
az workload-orchestration configuration set

# To get a configurations available at specified hierarchical entity.
az workload-orchestration configuration show

# Show the schema placeholder for a configuration template or solution template.
az workload-orchestration configuration schema show

# Create Context Resource.
az workload-orchestration context create

# Show current context information.
az workload-orchestration context current

# Delete Context Resource.
az workload-orchestration context delete

# List by specified resource group.
az workload-orchestration context list

# List by subscription.
az workload-orchestration context list-subscriptions

# Set current context using context ID.
az workload-orchestration context set

# Get Context Resource.
az workload-orchestration context show

# Use Context by name.
az workload-orchestration context use

# Place the CLI in a waiting state until a condition is met.
az workload-orchestration context wait

# Create Site Reference Resource.
az workload-orchestration context site-reference create

# Delete Site Reference Resource.
az workload-orchestration context site-reference delete

# List Site Reference Resources.
az workload-orchestration context site-reference list

# Get Site Reference Resource.
az workload-orchestration context site-reference show

# Place the CLI in a waiting state until a condition is met.
az workload-orchestration context site-reference wait

# Create new or updates existing Diagnostic resource.
az workload-orchestration diagnostic create

# Delete specified Diagnostic resource.
az workload-orchestration diagnostic delete

# List Diagnostics resources within an Azure subscription.
az workload-orchestration diagnostic list

# Get details of specified Diagnostic resource.
az workload-orchestration diagnostic show

# Update new or updates existing Diagnostic resource.
az workload-orchestration diagnostic update

# Place the CLI in a waiting state until a condition is met.
az workload-orchestration diagnostic wait

# Create a Schema Resource.
az workload-orchestration schema create

# Delete a Schema Resource.
az workload-orchestration schema delete

# List by subscription.
az workload-orchestration schema list

# Remove Schema Version Resource.
az workload-orchestration schema remove-version

# Get a Schema Resource.
az workload-orchestration schema show

# Place the CLI in a waiting state until a condition is met.
az workload-orchestration schema wait

# List by specified resource group.
az workload-orchestration schema version list

# Get a Schema Version Resource.
az workload-orchestration schema version show

# Place the CLI in a waiting state until a condition is met.
az workload-orchestration schema version wait

# Post request for bulk deploy.
az workload-orchestration solution-template bulk-deploy

# Post request for bulk publish.
az workload-orchestration solution-template bulk-publish

# Post request for bulk review.
az workload-orchestration solution-template bulk-review

# Create a Solution Template Resource.
az workload-orchestration solution-template create

# Delete a Solution Template Resource.
az workload-orchestration solution-template delete

# List by specified resource group.
az workload-orchestration solution-template list

# List by subscription.
az workload-orchestration solution-template list-subscriptions

# Remove Solution Template Version Resource.
az workload-orchestration solution-template remove-version

az workload-orchestration solution-template show

az workload-orchestration solution-template update-capabilities

az workload-orchestration solution-template wait

az workload-orchestration solution-template version list

# Get a Solution Template Version Resource.
az workload-orchestration solution-template version show

# Create a support bundle for troubleshooting workload orchestration issues.
az workload-orchestration support create-bundle

# Create a Target Resource.
az workload-orchestration target create

# Delete a Target Resource.
az workload-orchestration target delete

# Post request to install a solution.
az workload-orchestration target install

# List by Resource-Group.
az workload-orchestration target list

# List by subscription.
az workload-orchestration target list-subscriptions

# Post request to publish.
az workload-orchestration target publish

# Post request to remove solution version revision.
az workload-orchestration target remove-revision

# Post request to review configuration.
az workload-orchestration target review

az workload-orchestration target show

# List all solution instances of a solution deployed on a target.
```
