# az cognitiveservices

```bash
# Manage Azure Cognitive Services accounts.
az cognitiveservices account create

# Manage Azure Cognitive Services accounts.
az cognitiveservices account delete

# Manage Azure Cognitive Services accounts.
az cognitiveservices account list

# List soft-deleted Azure Cognitive Services accounts.
az cognitiveservices account list-deleted

# List all valid kinds for Azure Cognitive Services account.
az cognitiveservices account list-kinds

# Manage Azure Cognitive Services accounts.
az cognitiveservices account list-models

# Manage Azure Cognitive Services accounts.
az cognitiveservices account list-skus

# List usages for Azure Cognitive Services account.
az cognitiveservices account list-usage

# Purge a soft-deleted Azure Cognitive Services account.
az cognitiveservices account purge

# Recover a soft-deleted Azure Cognitive Services account.
az cognitiveservices account recover

# Manage Azure Cognitive Services accounts.
az cognitiveservices account show

# Show a soft-deleted Azure Cognitive Services account.
az cognitiveservices account show-deleted

# Manage Azure Cognitive Services accounts.
az cognitiveservices account update

# Create a commitment plan for Azure Cognitive Services account.
az cognitiveservices account commitment-plan create

# Delete a commitment plan from Azure Cognitive Services account.
az cognitiveservices account commitment-plan delete

# Show all commitment plans from Azure Cognitive Services account.
az cognitiveservices account commitment-plan list

# Show a commitment plan from Azure Cognitive Services account.
az cognitiveservices account commitment-plan show

# Create a connection.
az cognitiveservices account connection create

# Delete a connection.
az cognitiveservices account connection delete

# List all connections.
az cognitiveservices account connection list

# Show details of a connection.
az cognitiveservices account connection show

# Update a connection.
az cognitiveservices account connection update

# Create a deployment for Azure Cognitive Services account.
az cognitiveservices account deployment create

# Delete a deployment from Azure Cognitive Services account.
az cognitiveservices account deployment delete

# Show all deployments for Azure Cognitive Services account.
az cognitiveservices account deployment list

# Show a deployment for Azure Cognitive Services account.
az cognitiveservices account deployment show

# Assign an identity of a Cognitive Services account.
az cognitiveservices account identity assign

# Remove the identity from a Cognitive Services account.
az cognitiveservices account identity remove

# Show the identity of a Cognitive Services account.
az cognitiveservices account identity show

# Manage Azure Cognitive Services accounts.
az cognitiveservices account keys list

# Manage Azure Cognitive Services accounts.
az cognitiveservices account keys regenerate

# Add a network rule.
az cognitiveservices account network-rule add

# List network rules.
az cognitiveservices account network-rule list

# Remove a network rule.
az cognitiveservices account network-rule remove

# Create a project for Azure Cognitive Services account.
az cognitiveservices account project create

# Deletes a Cognitive Services project from the resource group.
az cognitiveservices account project delete

# Returns all the projects in a Cognitive Services account.
az cognitiveservices account project list

# Returns a Cognitive Services project specified by the parameters.
az cognitiveservices account project show

# Update a project for Azure Cognitive Services account.
az cognitiveservices account project update

# Create a connection for Azure Cognitive Services account.
az cognitiveservices account project connection create

# Delete Cognitive Services project connection by name.
az cognitiveservices account project connection delete

# Lists all the available Cognitive Services project connections under the specified project.
az cognitiveservices account project connection list

# Lists Cognitive Services project connection by name.
az cognitiveservices account project connection show

# Update a project connection.
az cognitiveservices account project connection update

# Create a new hosted agent from a container image or source code.
az cognitiveservices agent create

# Delete hosted agent version or all versions.
az cognitiveservices agent delete

# Delete hosted agent deployment.
az cognitiveservices agent delete-deployment

# List agents.
az cognitiveservices agent list

# List all versions of a hosted agent.
az cognitiveservices agent list-versions

# Show details of a hosted agent.
az cognitiveservices agent show

# Start a hosted agent deployment.
az cognitiveservices agent start

# Get the status of a hosted agent deployment.
az cognitiveservices agent status

# Stop a hosted agent deployment.
az cognitiveservices agent stop

# Update a hosted agent deployment.
az cognitiveservices agent update

# Show logs from a hosted agent container.
az cognitiveservices agent logs show

# Show all commitment tiers for Azure Cognitive Services.
az cognitiveservices commitment-tier list

# Show all models for Azure Cognitive Services.
az cognitiveservices model list

# Show all usages for Azure Cognitive Services.
az cognitiveservices usage list
```
