# az dnc

```bash
# Create a dnc controller.
az dnc controller create

# Delete the DNC controller.
az dnc controller delete

# Get details about the specified dnc controller.
az dnc controller show

# Place the CLI in a waiting state until a condition is met.
az dnc controller wait

# Create delegated subnet resource.
az dnc delegated-subnet-service create

# Delete dnc DelegatedSubnet.
az dnc delegated-subnet-service delete

# Get details about the specified dnc DelegatedSubnet Link.
az dnc delegated-subnet-service show

# Place the CLI in a waiting state until a condition is met.
az dnc delegated-subnet-service wait

# Create a orchestrator instance.
az dnc orchestrator-instance-service create

# Delete the Orchestrator Instance.
az dnc orchestrator-instance-service delete

# Get details about the orchestrator instance.
az dnc orchestrator-instance-service show

# Place the CLI in a waiting state until a condition is met.
az dnc orchestrator-instance-service wait
```
