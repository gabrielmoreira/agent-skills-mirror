# az providerhub

```bash
# Create the authorized application.
az providerhub authorized-application create

# Delete an authorized application.
az providerhub authorized-application delete

# List the list of the authorized applications in the provider namespace.
az providerhub authorized-application list

# Get the authorized application details.
az providerhub authorized-application show

# Update the authorized application.
az providerhub authorized-application update

# Place the CLI in a waiting state until a condition is met.
az providerhub authorized-application wait

# Create the rollout details.
az providerhub custom-rollout create

# Delete the custom rollout resource. Custom rollout must be in terminal state.
az providerhub custom-rollout delete

# List the list of the custom rollouts for the given provider.
az providerhub custom-rollout list

# Get the custom rollout details.
az providerhub custom-rollout show

# Stops or cancels the custom rollout, if in progress.
az providerhub custom-rollout stop

# Update the rollout details.
az providerhub custom-rollout update

# Place the CLI in a waiting state until a condition is met.
az providerhub custom-rollout wait

# Create the rollout details.
az providerhub default-rollout create

# Delete the rollout resource. Rollout must be in terminal state.
az providerhub default-rollout delete

# List the list of the rollouts for the given provider.
az providerhub default-rollout list

# Get the default rollout details.
az providerhub default-rollout show

# Stops or cancels the rollout, if in progress.
az providerhub default-rollout stop

# Update the rollout details.
az providerhub default-rollout update

# Place the CLI in a waiting state until a condition is met.
az providerhub default-rollout wait

# Checkin the manifest.
az providerhub manifest checkin

# Generates the manifest for the given provider.
az providerhub manifest generate

# Create a notification registration.
az providerhub notification-registration create

# Delete a notification registration.
az providerhub notification-registration delete

# List the list of the notification registrations for the given provider.
az providerhub notification-registration list

# Get the notification registration details.
az providerhub notification-registration show

# Update a notification registration.
az providerhub notification-registration update

# Create the operation supported by the given provider.
az providerhub operation create

# Delete an operation.
az providerhub operation delete

# List the operations supported by the given provider.
az providerhub operation list

# Update the operation supported by the given provider.
az providerhub operation update

# Create the provider registration.
az providerhub provider-registration create

# Delete a provider registration.
az providerhub provider-registration delete

# Generates the operations api for the given provider.
az providerhub provider-registration generate-operation

# List the list of the provider registrations in the subscription.
az providerhub provider-registration list

# Get the provider registration details.
az providerhub provider-registration show

# Update the provider registration.
az providerhub provider-registration update

# Place the CLI in a waiting state until a condition is met.
az providerhub provider-registration wait

# Deletes resources.
az providerhub provider-registration resource-action delete-resource

# Create a resource type.
az providerhub resource-type-registration create

# Delete a resource type.
az providerhub resource-type-registration delete

# List the list of the resource types for the given provider.
az providerhub resource-type-registration list

# Get a resource type details in the given subscription and provider.
az providerhub resource-type-registration show

# Update a resource type.
az providerhub resource-type-registration update

# Place the CLI in a waiting state until a condition is met.
az providerhub resource-type-registration wait

# Create the resource type skus in the given resource type.
az providerhub sku create

# Delete a resource type sku.
az providerhub sku delete

# List the list of skus for the given resource type.
az providerhub sku list

# Get the sku details for the given resource type and sku name.
az providerhub sku show

# Update the resource type skus in the given resource type.
az providerhub sku update

# Create the resource type skus in the given resource type.
az providerhub sku nested-resource-type-first create

# Delete a resource type sku.
az providerhub sku nested-resource-type-first delete

# List the list of skus for the given resource type.
az providerhub sku nested-resource-type-first list

# Get the sku details for the given resource type and sku name.
az providerhub sku nested-resource-type-first show

# Update the resource type skus in the given resource type.
az providerhub sku nested-resource-type-first update

# Create the resource type skus in the given resource type.
az providerhub sku nested-resource-type-second create

# Delete a resource type sku.
az providerhub sku nested-resource-type-second delete

# List the list of skus for the given resource type.
az providerhub sku nested-resource-type-second list

# Get the sku details for the given resource type and sku name.
az providerhub sku nested-resource-type-second show

# Update the resource type skus in the given resource type.
az providerhub sku nested-resource-type-second update

# Create the resource type skus in the given resource type.
az providerhub sku nested-resource-type-third create

# Delete a resource type sku.
az providerhub sku nested-resource-type-third delete

# List the list of skus for the given resource type.
az providerhub sku nested-resource-type-third list

# Get the sku details for the given resource type and sku name.
az providerhub sku nested-resource-type-third show

# Update the resource type skus in the given resource type.
az providerhub sku nested-resource-type-third update
```
