# az artifact-signing

```bash
# Checks that the artifact signing account name is valid and is not already in use.
az artifact-signing check-name-availability

# Create an artifact signing account.
az artifact-signing create

# Delete an artifact signing account.
az artifact-signing delete

# List artifact signing accounts within a subscription.
az artifact-signing list

# Get an artifact Signing Account.
az artifact-signing show

# Update an artifact signing account.
az artifact-signing update

# Place the CLI in a waiting state until a condition is met.
az artifact-signing wait

# Create a certificate profile.
az artifact-signing certificate-profile create

# Delete a certificate profile.
az artifact-signing certificate-profile delete

# List certificate profiles under an artifact signing account.
az artifact-signing certificate-profile list

# Get details of a certificate profile.
az artifact-signing certificate-profile show

# Place the CLI in a waiting state until a condition is met.
az artifact-signing certificate-profile wait
```
