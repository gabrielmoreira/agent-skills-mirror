# az managedcleanroom

```bash
# Adds a collaborator to a collaboration.
az managedcleanroom collaboration add-collaborator

# Create a collaboration.
az managedcleanroom collaboration create

# Delete a collaboration.
az managedcleanroom collaboration delete

# Enables a workload on a collaboration.
az managedcleanroom collaboration enable-workload

# List collaborations in a subscription.
az managedcleanroom collaboration list

# Pauses collaboration.
az managedcleanroom collaboration pause

# Recovers collaboration.
az managedcleanroom collaboration recover

# Resumes collaboration.
az managedcleanroom collaboration resume

# Get a collaboration.
az managedcleanroom collaboration show

# Update a collaboration.
az managedcleanroom collaboration update

# Place the CLI in a waiting state until a condition is met.
az managedcleanroom collaboration wait

# Create a consortium view.
az managedcleanroom consortium-view create

# Delete a consortium view.
az managedcleanroom consortium-view delete

# List consortium views in a subscription.
az managedcleanroom consortium-view list

# Get a consortium view.
az managedcleanroom consortium-view show

# Update a consortium view.
az managedcleanroom consortium-view update

# Place the CLI in a waiting state until a condition is met.
az managedcleanroom consortium-view wait

# List contracts in a consortium view.
az managedcleanroom consortium-view contract list

# Generates and proposes contract template.
az managedcleanroom consortium-view contract propose-template

# Get a contract.
az managedcleanroom consortium-view contract show

# Create a consortium.
az managedcleanroom consortium create

# Delete a consortium.
az managedcleanroom consortium delete

# List consortiums in a subscription.
az managedcleanroom consortium list

# Pauses consortium.
az managedcleanroom consortium pause

# Recovers consortium.
az managedcleanroom consortium recover

# Resumes consortium.
az managedcleanroom consortium resume

# Get a consortium.
az managedcleanroom consortium show

# Update a consortium.
az managedcleanroom consortium update

# Place the CLI in a waiting state until a condition is met.
az managedcleanroom consortium wait

# Configure or view Analytics Frontend API settings.
az managedcleanroom frontend configure

# Authenticate using Microsoft device code flow.
az managedcleanroom frontend login

# Logout and clear cached credentials.
az managedcleanroom frontend logout

# Get comprehensive attestation report.
az managedcleanroom frontend report

# Show collaboration details.
az managedcleanroom frontend show

# Get cleanroom policy.
az managedcleanroom frontend analytics cleanroompolicy

# Show analytics information.
az managedcleanroom frontend analytics show

# List audit events for a collaboration.
az managedcleanroom frontend analytics auditevent list

# List datasets for a collaboration.
az managedcleanroom frontend analytics dataset list

# Publish a dataset to the collaboration.
az managedcleanroom frontend analytics dataset publish

# List queries that use a specific dataset.
az managedcleanroom frontend analytics dataset queries

# Show dataset details.
az managedcleanroom frontend analytics dataset show

# List queries for a collaboration.
az managedcleanroom frontend analytics query list

# Publish a query to the collaboration.
az managedcleanroom frontend analytics query publish

# Run a query in the collaboration.
az managedcleanroom frontend analytics query run

# Show query details.
az managedcleanroom frontend analytics query show

# Vote on a query (accept or reject).
az managedcleanroom frontend analytics query vote

# List query run history.
az managedcleanroom frontend analytics query runhistory list

# Show details of a query run result.
az managedcleanroom frontend analytics query runresult show

# Set an analytics secret.
az managedcleanroom frontend analytics secret set

# List all collaborations.
az managedcleanroom frontend collaboration list

# Check consent document status.
az managedcleanroom frontend consent check

# Set consent document action.
az managedcleanroom frontend consent set

# Accept an invitation.
az managedcleanroom frontend invitation accept

# List invitations for a collaboration.
az managedcleanroom frontend invitation list

# Show invitation details.
az managedcleanroom frontend invitation show

# Get OIDC signing keys (JWKS).
az managedcleanroom frontend oidc keys

# Set OIDC issuer URL for a collaboration.
az managedcleanroom frontend oidc set-issuer-url

# Show OIDC issuer information.
az managedcleanroom frontend oidc issuerinfo show
```
