# az mesh

```bash
# Delete a Service Fabric Mesh application.
az mesh app delete

# List Service Fabric Mesh applications.
az mesh app list

# Get the details of a Service Fabric Mesh application.
az mesh app show

# Examine the logs for a codepackage.
az mesh code-package-log get

# Create a Service Fabric Mesh application.
az mesh deployment create

# Delete a gateway resource.
az mesh gateway delete

# List gateway resources.
az mesh gateway list

# Get the details of a gateway.
az mesh gateway show

# Generate a Service Fabric Mesh ARM deployment template from input yaml.
az mesh generate armtemplate

# Delete a network.
az mesh network delete

# List networks.
az mesh network list

# Get the details of a network.
az mesh network show

# Delete a secret.
az mesh secret delete

# List Secrets.
az mesh secret list

# Get the details of a secret.
az mesh secret show

# Delete a secret version.
az mesh secretvalue delete

# List Secrets versions.
az mesh secretvalue list

# Get the details of a secret value.
az mesh secretvalue show

# List the details of service replicas.
az mesh service-replica list

# Gets the given replica of the service of an application.
az mesh service-replica show

# List all the service resources.
az mesh service list

# Get the details of a service.
az mesh service show

# Create a volume.
az mesh volume create

# Delete a volume.
az mesh volume delete

# List volumes.
az mesh volume list

# Get the details of a volume.
az mesh volume show
```
