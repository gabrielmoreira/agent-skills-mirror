# az edge-action

```bash
# Create EdgeAction resource.
az edge-action create

# Delete EdgeAction resource.
az edge-action delete

# List EdgeAction resources by subscription ID.
az edge-action list

# Get EdgeAction resource.
az edge-action show

# Update EdgeAction resource.
az edge-action update

# Place the CLI in a waiting state until a condition is met.
az edge-action wait

# Create EdgeActionExecutionFilter resource.
az edge-action execution-filter create

# Delete EdgeActionExecutionFilter resource.
az edge-action execution-filter delete

# List EdgeActionExecutionFilter resources by EdgeAction.
az edge-action execution-filter list

# Get EdgeActionExecutionFilter resource.
az edge-action execution-filter show

# Update EdgeActionExecutionFilter resource.
az edge-action execution-filter update

# Place the CLI in a waiting state until a condition is met.
az edge-action execution-filter wait

# Create EdgeActionVersion version.
az edge-action version create

# Delete EdgeActionVersion resource.
az edge-action version delete

# Deploy edge action version code from a JavaScript or zip file.
az edge-action version deploy-from-file

# A long-running operation to deploy versioncode to EdgeActionVersion resource.
az edge-action version deploy-version-code

# A long-running operation to get versioncode deployed to EdgeActionVersion resource.
az edge-action version get-version-code

# List EdgeActionVersion resources by EdgeAction.
az edge-action version list

# Get EdgeActionVersion resource.
az edge-action version show

# A long-running resource action.
az edge-action version swap-default

# Update EdgeActionVersion version.
az edge-action version update

# Place the CLI in a waiting state until a condition is met.
az edge-action version wait
```
