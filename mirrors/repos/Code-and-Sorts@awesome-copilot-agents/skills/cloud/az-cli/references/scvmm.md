# az scvmm

```bash
# Create availabilty set resource.
az scvmm avset create

# Delete availabilty set resource.
az scvmm avset delete

# Retrieve a list of availabilty sets.
az scvmm avset list

# Get details of an availabilty set.
az scvmm avset show

# Update availabilty set resource.
az scvmm avset update

# Place the CLI in a waiting state until a condition of the availability set is met.
az scvmm avset wait

# Create cloud resource.
az scvmm cloud create

# Delete cloud resource.
az scvmm cloud delete

# Retrieve a list of clouds.
az scvmm cloud list

# Get details of a cloud.
az scvmm cloud show

# Update cloud resource.
az scvmm cloud update

# Place the CLI in a waiting state until a condition of the cloud is met.
az scvmm cloud wait

# Create virtual-network resource.
az scvmm virtual-network create

# Delete virtual-network resource.
az scvmm virtual-network delete

# Retrieve a list of virtual-networks.
az scvmm virtual-network list

# Get details of a virtual-network.
az scvmm virtual-network show

# Update virtual-network resource.
az scvmm virtual-network update

# Place the CLI in a waiting state until a condition of the virtual-network is met.
az scvmm virtual-network wait

# Create vm-template resource.
az scvmm vm-template create

# Delete vm-template resource.
az scvmm vm-template delete

# Retrieve a list of vm-templates.
az scvmm vm-template list

# Get details of a vm-template.
az scvmm vm-template show

# Update vm-template resource.
az scvmm vm-template update

# Place the CLI in a waiting state until a condition of the vm-template is met.
az scvmm vm-template wait

# Create VM resource.
az scvmm vm create

# Create a VM checkpoint.
az scvmm vm create-checkpoint

# Create VMInstance resource(s) from existing Microsoft.HybridCompute machines.
az scvmm vm create-from-machines

# Delete VM resource.
az scvmm vm delete

# Delete the specified VM checkpoint.
az scvmm vm delete-checkpoint

# Restart a VM.
az scvmm vm restart

# Restore VM checkpoint.
az scvmm vm restore-checkpoint

# Get details of an VM.
az scvmm vm show

# Start a VM.
az scvmm vm start

# Stop a VM.
az scvmm vm stop

# Update a VM. Management of VM disks and NICs are not using this subcommand. There are separate subcommands for the same.
az scvmm vm update

# Place the CLI in a waiting state until a condition of the vm is met.
az scvmm vm wait

# Add a virtual disk to a virtual machine.
az scvmm vm disk add

# Delete disks of a virtual machine.
az scvmm vm disk delete

# Retrieve the list of disks present in a VM.
az scvmm vm disk list

# Get the details of a disk present in a VM.
az scvmm vm disk show

# Update a disk of a VM.
az scvmm vm disk update

# Place the CLI in a waiting state until a condition of the vm disk is met.
az scvmm vm disk wait

# The operation to create the extension.
az scvmm vm extension create

# The operation to delete the extension.
az scvmm vm extension delete

# The operation to get all extensions of a non-Azure vm.
az scvmm vm extension list

# The operation to get the extension.
az scvmm vm extension show

# The operation to update the extension.
az scvmm vm extension update

# Enable guest agent on the vm.
az scvmm vm guest-agent enable

# Get details of a guest agent by guest agent name, resource-group and vm name.
az scvmm vm guest-agent show

# Add a network interface card to a virtual machine.
az scvmm vm nic add

# Delete NICs of a virtual machine.
az scvmm vm nic delete

# Retrieve the list of NICs present in a VM.
az scvmm vm nic list

# Get the details of a NIC present in a VM.
az scvmm vm nic show

# Update a NIC of a VM.
az scvmm vm nic update

# Place the CLI in a waiting state until a condition of the vm nic is met.
az scvmm vm nic wait

# Create a vmmserver resource.
az scvmm vmmserver connect

# Delete a vmmserver resource.
az scvmm vmmserver delete

# Retrieve a list of vmmservers.
az scvmm vmmserver list

# Get details of a vmmserver.
az scvmm vmmserver show

# Update vmmserver resource.
az scvmm vmmserver update

# Place the CLI in a waiting state until a condition of the vmmserver is met.
az scvmm vmmserver wait

# Retrieve a list of inventory items present in a vmmserver.
az scvmm vmmserver inventory-item list

# Get details of an inventory item present in a vmmserver.
az scvmm vmmserver inventory-item show
```
