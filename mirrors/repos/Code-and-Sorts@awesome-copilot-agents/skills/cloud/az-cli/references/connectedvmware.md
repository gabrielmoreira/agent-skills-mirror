# az connectedvmware

```bash
# Create a cluster resource.
az connectedvmware cluster create

# Delete cluster resource.
az connectedvmware cluster delete

# Retrieve a list of cluster of given resource group.
az connectedvmware cluster list

# Get details of a cluster by id, resource-group, cluster name, or subscription.
az connectedvmware cluster show

# Create a datastore resource.
az connectedvmware datastore create

# Delete datastore resource.

az connectedvmware datastore show

az connectedvmware host create

az connectedvmware host delete

# Retrieve a list of host of given resource group.
az connectedvmware host list

# Get details of a host by id, resource-group, host name, or subscription.
az connectedvmware host show

az connectedvmware resource-pool delete

# Retrieve a list of resource pool of given resource group.
az connectedvmware resource-pool list

# Get details of a resource pool by id, resource-group, resource pool name, or subscription.
az connectedvmware resource-pool show

az connectedvmware vcenter delete

az connectedvmware vcenter list

az connectedvmware vm-template list

az connectedvmware vm-template show

az connectedvmware vm create

az connectedvmware vm delete

az connectedvmware vm restart

az connectedvmware vm show

az connectedvmware vm start

az connectedvmware vm update

# Add virtual disk to a virtual machine.
az connectedvmware vm disk add

# Delete virtual disks to a virtual machine.
az connectedvmware vm disk delete

# Retrieve a list of vm disk from given resource group name and vm name.
az connectedvmware vm disk list

# Get details of a vm disk by it's name, resource-group and vm name.
az connectedvmware vm disk show

# Update virtual disk to a virtual machine.
az connectedvmware vm disk update

# The operation to create the extension.
az connectedvmware vm extension create

# The operation to delete the extension.
az connectedvmware vm extension delete

# The operation to get all extensions of a non-Azure vm.
az connectedvmware vm extension list

# The operation to get the extension.
az connectedvmware vm extension show

# The operation to update the extension.
az connectedvmware vm extension update

# Enable guest agent on the vm.

# Get details of a guest agent by guest agent name, resource-group and vm name.
```
