# az stack-hci

```bash
# Add consent time for default extensions and initiate extensions installation.
az stack-hci arc-setting consent-and-install-default-extension

# Create arc setting for HCI cluster.
az stack-hci arc-setting create

# Create AAD identity for arc settings.
az stack-hci arc-setting create-identity

# Delete arc setting resource details of HCI cluster.
az stack-hci arc-setting delete

# Generate password for arc settings.
az stack-hci arc-setting generate-password

# Initialize ARC Disable process on the cluster.
az stack-hci arc-setting initialize-disable-proces

# List arc setting resources of HCI cluster.
az stack-hci arc-setting list

# Get arc setting resource details of HCI cluster.
az stack-hci arc-setting show

# Update arc setting for HCI cluster.
az stack-hci arc-setting update

# Place the CLI in a waiting state until a condition is met.
az stack-hci arc-setting wait

# Create an HCI cluster.
az stack-hci cluster create

# Create cluster identity.
az stack-hci cluster create-identity

# Delete an HCI cluster.
az stack-hci cluster delete

# Extend Software Assurance Benefit to a cluster.
az stack-hci cluster extend-software-assurance-benefit

# List all HCI clusters in a subscription.
az stack-hci cluster list

# Get HCI cluster.
az stack-hci cluster show

# Update an HCI cluster.
az stack-hci cluster update

# Place the CLI in a waiting state until a condition is met.
az stack-hci cluster wait

# Assign identities.
az stack-hci cluster identity assign

# Remove identities.
az stack-hci cluster identity remove

# Create extension for HCI cluster.
az stack-hci extension create

# Delete particular arc extension of HCI Cluster.
az stack-hci extension delete

# List all extensions under arc setting resource.
az stack-hci extension list

# Get particular arc extension of HCI cluster.
az stack-hci extension show

# Place the CLI in a waiting state until a condition is met.
az stack-hci extension wait
```
