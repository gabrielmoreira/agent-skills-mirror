# az offazure

```bash
# Get all clusters on the on-premise site.
az offazure hyperv cluster list

# Get the details of a Hyper-V cluster.
az offazure hyperv cluster show

# Get all hosts on the on-premise site.
az offazure hyperv host list

# Get the details of a Hyper-V host.
az offazure hyperv host show

# List all machines on the on-premise site.
az offazure hyperv machine list

# Get the details of a machine.
az offazure hyperv machine show

# List all run-as-accounts on the on-premise site.
az offazure hyperv run-as-account list

# Get the details of a run-as-account.
az offazure hyperv run-as-account show

# Create a Hyper-V site.
az offazure hyperv site create

# Delete a Hyper-V site.
az offazure hyperv site delete

# Get the details of a site.
az offazure hyperv site show

# List all machines on the on-premise site.
az offazure vmware machine list

# Get the details of a machine.
az offazure vmware machine show

# List all run-as-accounts on the on-premise site.
az offazure vmware run-as-account list

# Get the details of a run-as-account.
az offazure vmware run-as-account show

# Create a site for VMware resources.
az offazure vmware site create

# Delete a VMware site.
az offazure vmware site delete

# Get the details of a VMware site.
az offazure vmware site show

# List all vCenters on the on-premise site.
az offazure vmware vcenter list

# Get the details of a vCenter.
az offazure vmware vcenter show
```
