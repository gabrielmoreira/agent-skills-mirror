# az durabletask

```bash
# Create a Retention Policy.
az durabletask retention-policy create

# Delete a Retention Policy.
az durabletask retention-policy delete

# Get a Retention Policy.
az durabletask retention-policy show

# Place the CLI in a waiting state until a condition is met.
az durabletask retention-policy wait

# [Experimental] Attach a Durable Task scheduler to a Function App or Container App.
az durabletask scheduler attach

# Create a Scheduler.
az durabletask scheduler create

# Delete a Scheduler.
az durabletask scheduler delete

# List Schedulers by subscription.
az durabletask scheduler list

# Get a Scheduler.
az durabletask scheduler show

# Update a Scheduler.
az durabletask scheduler update

# Place the CLI in a waiting state until a condition is met.
az durabletask scheduler wait

# Create a Task Hub.
az durabletask taskhub create

# Delete a Task Hub.
az durabletask taskhub delete

# List Task Hubs.
az durabletask taskhub list

# Get a Task Hub.
az durabletask taskhub show

# Place the CLI in a waiting state until a condition is met.
az durabletask taskhub wait
```
