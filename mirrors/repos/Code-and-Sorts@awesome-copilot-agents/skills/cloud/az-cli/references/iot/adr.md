# az iot adr

```bash
# Create a Device Registry namespace.
az iot adr ns create

# Delete a Device Registry namespace.
az iot adr ns delete

# List Device Registry namespaces.
az iot adr ns list

# Show details of a Device Registry namespace.
az iot adr ns show

# Update a Device Registry namespace.
az iot adr ns update

# Create credentials for a Device Registry namespace.
az iot adr ns credential create

# Delete a credential for a Device Registry namespace.
az iot adr ns credential delete

# Show credentials for a Device Registry namespace.
az iot adr ns credential show

# Synchronize Device Registry credentials to linked Iot Hubs.
az iot adr ns credential sync

# Create a policy for a Device Registry namespace.
az iot adr ns policy create

# Delete a policy for a Device Registry namespace.
az iot adr ns policy delete

# List policies for Device Registry namespaces.
az iot adr ns policy list

# Show a policy for a Device Registry namespace.
az iot adr ns policy show

# Update a policy for a Device Registry namespace.
az iot adr ns policy update
```
